   สร้างโหมด venv ลงในโปรเจค 
	python -m venv {NameVenv}
   เข้าใช้งานโหมด venv
	venv\Scripts\activate
   ลงส่วนเสริมใน pip ที่ต้องใช้งาน
   สร้างโฟลเดอร์ชื่อ app -> สร้างไฟล์ชื่อ __init__.py 
 app/__init__.py 
        ภายในไฟล์ __init__.py
	from flask import Flask
	app = Flask(__name__)
	from app import routes

   สร้างไฟล์ชื่อ routes.py
 app/routes.py
        ภายในไฟล์ routes.py
	from app import app

	@app.route('/')
	@app.route('/index')
	def index():
    		return "Hello Flask World!"

    สร้างไฟล์ชื่อ project_script.py

         ภายในไฟล์ project_script.py
